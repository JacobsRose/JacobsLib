local list = {}
local Hooks = {
    On = function (name, func)

        if list[name] == nil then --If the exact hook isn`t initialized yet
            list[name] = {}
            hook.add(name, "", function(...)
                for k,v in pairs(list[name]) do
                    k(...)
                end
            end)
        end

        list[name][func] = true --Connecting new func

    end,
    Off = function (name, func)

        list[name][func] = nil

        if table.count(list[name]) == 0 then
            list[name] = nil
            hook.remove(name, "")
        end
    end
}
return Hooks
