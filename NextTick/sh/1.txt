local Async = require("../../Async/sh/2.txt")
local Hooks = require("../../Hooks/sh/1.txt")

local NextTick = Async(function(callback)

    local function func()
        Hooks.Off("tick", func)
        callback()
    end
    Hooks.On("tick", func)
end)
return NextTick
